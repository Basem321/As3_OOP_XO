56 int minimax(vector<vector<char>>& board, int depth, bool is_maximizing, int alpha, int beta) {
57 if (check_win(board, this->symbol)) return 10 - depth;
58 if (check_win(board, get_opponent_symbol())) return depth - 10;
59 vector<pair<int, int>> moves = get_valid_moves(board);
60 if (moves.empty()) return 0;
61
62 if (is_maximizing) {
63 int max_eval = numeric_limits<int>::min();
64 for (auto& move : moves) {
65 board[move.first][move.second] = this->symbol;
66 int eval = minimax(board, depth + 1, false, alpha, beta);
67 board[move.first][move.second] = '.';
68 max_eval = max(max_eval, eval);
69 alpha = max(alpha, eval);
70 if (beta <= alpha) break;
71 }
72 return max_eval;
73 }
74 else {
75 int min_eval = numeric_limits<int>::max();
76 char opp = get_opponent_symbol();
77 for (auto& move : moves) {
78 board[move.first][move.second] = opp;
79 int eval = minimax(board, depth + 1, true, alpha, beta);
80 board[move.first][move.second] = '.';
81 min_eval = min(min_eval, eval);
82 beta = min(beta, eval);
