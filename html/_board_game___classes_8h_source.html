271template <typename T>
272class GameManager {
273 Board<T>* boardPtr;
274 Player<T>* players[2];
275 UI<T>* ui;
276
277public:
281 GameManager(Board<T>* b, Player<T>* p[2], UI<T>* u)
282 : boardPtr(b), ui(u) {
283 players[0] = p[0];
284 players[1] = p[1];
285 players[0]->set_board_ptr(b);
286 players[1]->set_board_ptr(b);
287 }
288
292 void run() {
293 ui->display_board_matrix(boardPtr->get_board_matrix());
294 Player<T>* currentPlayer = players[0];
295
296 while (true) {
297 for (int i : {0, 1}) {
298 currentPlayer = players[i];
299 Move<T>* move = ui->get_move(currentPlayer);
300
301 while (!boardPtr->update_board(move))
302 move = ui->get_move(currentPlayer);
303
304 ui->display_board_matrix(boardPtr->get_board_matrix());
305
306 if (boardPtr->is_win(currentPlayer)) {
307 ui->display_message(currentPlayer->get_name() + " wins!");
308 return;
309 }
310 if (boardPtr->is_lose(currentPlayer)) {
311 ui->display_message(players[1 - i]->get_name() + " wins!");
312 return;
313 }
314 if (boardPtr->is_draw(currentPlayer)) {
315 ui->display_message("Draw!");
316 return;
317 }
318 }
319 }
320 }
321};
